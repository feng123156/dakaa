<script> 

const user = {role:"",name:"",cls:"",score:0,stage:""}; 

const levels = ["初级","中级","高级"]; 

const cases = [ 

[ 

{title:"初级题1",desc:"同学A刚学会一点PPT，就觉得自己比老师还厉害。请问他处在达克效应的哪个阶段？为什么？",std:"愚昧之山；刚入门缺乏认知，高估自己。",score:15,check:(a)=>/愚昧|山/.test(a)}, 

{title:"初级题2",desc:"同学B第一次考试考很好，第二次惨败，突然觉得自己什么都不会。他处在哪个阶段？",std:"绝望之谷；从高估跌落到觉醒，自信骤降。",score:15,check:(a)=>/绝望|谷/.test(a)} 

], 

[ 

{title:"中级题1",desc:"举例说明生活中‘愚昧之山’现象，并写出1个克服方法。",std:"例：新手觉得自己很懂；策略：寻求反馈、客观测试、持续学习。",score:15,check:(a)=>a.length>20}, 

{title:"中级题2",desc:"为什么高手容易低估自己？用达克效应解释。",std:"高手以为别人和自己一样强，产生偏差低估。",score:15,check:(a)=>/低估|以为别人/.test(a)} 

], 

[ 

{title:"高级题1",desc:"完整描述四阶段变化，并说明对学习的启示。",std:"愚昧之山→绝望之谷→开悟之坡→平稳高原；启示：保持谦卑、持续学习、接受反馈。",score:15,check:(a)=>a.length>40}, 

{title:"高级题2",desc:"分析‘元认知’在达克效应中的作用。",std:"元认知弱→无法自知（愚昧）；元认知强→能正确评估（开悟）。",score:15,check:(a)=>/元认知/.test(a)} 

] 

]; 

 

let lv=0,idx=0; 

const data = []; 

 

function login(role){ 

user.role=role;user.name=document.getElementById("name").value||"学生";user.cls=document.getElementById("cls").value||"一班"; 

document.getElementById("login").classList.add("hide"); 

if(role=="student"){document.getElementById("student").classList.remove("hide");renderCase()} 

else{document.getElementById("teacher").classList.remove("hide");renderTeacher()} 

} 

 

function renderCase(){ 

const c=cases[lv][idx]; 

document.getElementById("caseTip").textContent=`当前：${levels[lv]} 第${idx+1}/2题`; 

document.getElementById("qTitle").textContent=c.title; 

document.getElementById("qDesc").textContent=c.desc; 

document.getElementById("ans").value=""; 

document.getElementById("result").classList.add("hide"); 

} 

 

function setLevel(n){lv=n;idx=0;renderCase()} 

function nextCase(){idx=(idx+1)%cases[lv].length;renderCase()} 

 

function check(){ 

const c=cases[lv][idx]; 

const a=document.getElementById("ans").value.trim(); 

const ok=c.check(a); 

const s=ok?c.score:0; 

user.score=s; 

if(lv==0)user.stage=s>=15?"愚昧之山":"已觉醒"; 

if(lv==1)user.stage=s>=15?"开悟之坡":"进步中"; 

if(lv==2)user.stage=s>=15?"平稳高原":"优秀学习者"; 

 

const fb=document.getElementById("fb"); 

const scoreTxt=document.getElementById("scoreTxt"); 

const stage=document.getElementById("stage"); 

const stdAns=document.getElementById("stdAns"); 

const result=document.getElementById("result"); 

 

result.classList.remove("hide"); 

fb.className=ok?"feedback correct":"feedback wrong"; 

fb.textContent=ok?"✅ 回答正确":"❌ 回答有误"; 

scoreTxt.textContent=`得分：${s}/15`; 

stage.textContent=`认知阶段判定：${user.stage}`; 

stdAns.textContent=c.std; 

 

if(user.role=="student"){data.push({cls:user.cls,name:user.name,score:s,stage:user.stage})} 

} 

 

function clearAns(){document.getElementById("ans").value="";document.getElementById("result").classList.add("hide")} 

function renderTeacher(){ 

const t=document.getElementById("data");t.innerHTML=""; 

data.forEach(i=>{const tr=document.createElement("tr"); 

tr.innerHTML=`<td>${i.cls}</td><td>${i.name}</td><td>${i.score}</td><td>${i.stage}</td>`;t.appendChild(tr)}) 

}
